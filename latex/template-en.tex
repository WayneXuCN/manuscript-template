\subsection{Detailed Proof of Theorem 1}

\begin{theorem}[Convergence Theorem]
\label{thm:convergence}
    Assume the objective function $f(x)$ is $L$-smooth, and the learning rate $\eta \leq \frac{1}{L}$. Then Algorithm 1 guarantees convergence after $T$ iterations:
    \begin{equation}
        \min_{t \in [T]} \|\nabla f(x_t)\|^2 \leq \frac{2(f(x_0) - f^*)}{\eta T}
    \end{equation}
\end{theorem}

\begin{proof}
    By the definition of $L$-smoothness, for any $x, y$ we have:
    \begin{equation}
        f(y) \leq f(x) + \nabla f(x)^\top(y-x) + \frac{L}{2}\|y-x\|^2
    \end{equation}
    
    Let $y = x_{t+1} = x_t - \eta \nabla f(x_t)$, substituting into the above inequality:
    \begin{align}
        f(x_{t+1}) &\leq f(x_t) - \eta \|\nabla f(x_t)\|^2 + \frac{L\eta^2}{2}\|\nabla f(x_t)\|^2 \\
        &= f(x_t) - \eta\left(1 - \frac{L\eta}{2}\right)\|\nabla f(x_t)\|^2
    \end{align}
    
    When $\eta \leq \frac{1}{L}$, we have $1 - \frac{L\eta}{2} \geq \frac{1}{2}$, therefore:
    \begin{equation}
        f(x_{t+1}) \leq f(x_t) - \frac{\eta}{2}\|\nabla f(x_t)\|^2
    \end{equation}
    
    Summing over $t = 0, 1, \ldots, T-1$:
    \begin{equation}
        f(x_T) \leq f(x_0) - \frac{\eta}{2}\sum_{t=0}^{T-1}\|\nabla f(x_t)\|^2
    \end{equation}
    
    Rearranging and noting that $f(x_T) \geq f^*$ completes the proof.
\end{proof}
